/**
 * Generic Stack implementation using array in Java
 * Supports basic stack operations (push, pop, peek, size, isEmpty)
 * Maximum capacity is fixed (1000 elements)
 * Generic type T allows any object type to be stored
 */
class Stack<T> {

    private static final int MaxSize = 1000; // Maximum capacity of the stack
    private T[] array;                        // Array to store stack elements
    private int top;                           // Index of the top element

    /**
     * Constructor: initializes an empty stack
     * Uses Object array and casts it to generic T[]
     */
    @SuppressWarnings("unchecked")
    public Stack() {
        array = (T[]) new Object[MaxSize];
        top = -1; // stack is empty initially
    }

    /**
     * Checks whether the stack is empty
     * @return true if no elements in the stack, false otherwise
     */
    boolean isEmpty() {
        return top == -1;
    }

    /**
     * Checks whether the stack is full
     * @return true if stack reached maximum capacity
     */
    boolean isFull() {
        return top == MaxSize - 1;
    }

    /**
     * Adds a new element to the top of the stack
     * @param ele element to be added
     * @throws RuntimeException if stack is full
     */
    void push(T ele) {
        if (isFull()) {
            throw new RuntimeException("Stack is Full");
        }
        array[++top] = ele; // increment top and insert element
    }

    /**
     * Removes and returns the top element of the stack
     * @return element at the top of the stack
     * @throws RuntimeException if stack is empty
     */
    T pop() {
        if (isEmpty()) {
            throw new RuntimeException("Stack is Empty");
        }
        return array[top--]; // return top element and decrement top
    }

    /**
     * Returns the top element without removing it
     * @return element at the top
     * @throws RuntimeException if stack is empty
     */
    T peek() {
        if (isEmpty()) {
            throw new RuntimeException("Stack is Empty");
        }
        return array[top];
    }

    /**
     * Returns the number of elements currently in the stack
     * @return size of the stack
     */
    int size() {
        return top + 1;
    }

    /**
     * Prints all elements of the stack from top to bottom
     * Example output: [top, ..., bottom]
     */
    void print() {
        if (isEmpty()) {
            System.out.println("[]");
            return;
        }

        System.out.print("[");
        for (int i = top; i >= 0; i--) {
            System.out.print(array[i]);
            if (i != 0) System.out.print(", ");
        }
        System.out.println("]");
    }
}

public class Main {
    public static void main(String[] args) {

        // Create a stack of integers
        Stack<Integer> stack = new Stack<>();

        // Push elements to the stack
        stack.push(24);
        stack.push(25);
        stack.push(26);

        // Print current stack: [26, 25, 24]
        stack.print();

        // Peek the top element: 26
        System.out.println(stack.peek());

        // Pop the top element: 26
        System.out.println(stack.pop());

        // Check if stack is empty: false
        System.out.println(stack.isEmpty());

        // Get current size: 2
        System.out.println(stack.size());
    }
}
